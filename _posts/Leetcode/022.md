class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        ListNode* slow;
        ListNode* fast;
        slow = fast = head;
        //此处要判断fast的后继是否为空
        while(fast!=NULL&&fast->next!=NULL){
            slow = slow->next;
            fast = fast->next->next;
            if(slow == fast){
                ListNode* p1 = head;
                while(p1!=slow){
                    p1 = p1->next;
                    slow = slow->next;
                }
                return p1;
            }
        }
        return NULL;
    }
};
